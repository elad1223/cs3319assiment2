CREATE TABLE product(productID INT NOT NULL,description VARCHAR(20),cost FLOAT,quantity INT,PREMIRE KEY(productID));

CREATE TABLE agent(agentID INT, NOT NULL,firstname VARCHAR(30),lastname VARCHAR(30),city VARCHAR(15),commissioPercent INT,PREMIRE KEY(agentID));

CREATE TABLE customer(customerID INT NOT NULL,firstname VARCHAR(30),lastname VARCHAR(30), city VARCHAR(15),phoneNumber INT,agentID VARCHAR(2),PREMISE KEY(customerID),FOREIGN KEY(agentID) REFERENCES agent(agentID));
